#include<stdio.h>
                #include<stdlib.h>

                struct stack{
                    int size;
                    int top;
                    int* arr;
                };

                int main(){
                    struct stack* stack;
                    stack = (struct stack*)malloc(sizeof(struct stack));
                    stack->size = 20;
                    stack->top = -1;
                    stack->arr = (int*)malloc(stack->size*sizeof(int));

                    return 0;
                }
